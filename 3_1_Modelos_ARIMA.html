# Funcin para cargar y preparar datos
def load_and_prepare_data(file_path, stock_name):
    """
    Carga datos de un archivo Excel y crea un ndice DatetimeIndex.
    Args:
        file_path (str): Ruta al archivo Excel
        stock_name (str): Nombre de la accin
    Returns:
        pd.DataFrame: DataFrame con ndice DatetimeIndex
    """
    if not os.path.exists(file_path):
        raise FileNotFoundError(f"El archivo {file_path} no se encuentra en el directorio actual.")
    df = pd.read_excel(file_path)
    # Convertir Local Date a string en formato 'YYYY-MM-DD'
    df['DateStr'] = df['Local Date'].dt.strftime('%Y-%m-%d')
    # Convertir Local Time a string en formato 'HH:MM:SS'
    df['TimeStr'] = df['Local Time'].apply(lambda x: x.strftime('%H:%M:%S') if isinstance(x, (pd.Timestamp, datetime, pd._libs.tslibs.nattype.NaTType)) else str(x))
    # Combinar DateStr y TimeStr para crear Datetime
    df['Datetime'] = pd.to_datetime(df['DateStr'] + ' ' + df['TimeStr'])
    df.drop(['DateStr', 'TimeStr'], axis=1, inplace=True)
    df.set_index('Datetime', inplace=True)
    df = df[['Close']].rename(columns={'Close': stock_name})
    return df

# Funcin para graficar serie de tiempo
def plot_time_series(df, stock_name):
    """
    Grafica la serie de precios de cierre y la muestra en Colab.
    """
    plt.figure(figsize=(12, 6))
    plt.plot(df.index, df[stock_name], label=f'{stock_name} Close Price')
    plt.title(f'{stock_name} Stock Price (5-min Intraday)')
    plt.xlabel('Time')
    plt.ylabel('Price')
    plt.legend()
    plt.grid(True)
    plt.savefig(f'{stock_name}_time_series.png')
    display(Image(filename=f'{stock_name}_time_series.png'))
    plt.close()

# Funcin para prueba de estacionariedad (ADF)
def test_stationarity(series, stock_name):
    """
    Realiza la prueba ADF para verificar estacionariedad.
    Returns:
        dict: Resultados de la prueba ADF
    """
    result = adfuller(series.dropna())
    print(f'\nADF Test for {stock_name}:')
    print(f'ADF Statistic: {result[0]:.4f}')
    print(f'p-value: {result[1]:.4f}')
    print('Critical Values:')
    for key, value in result[4].items():
        print(f'\t{key}: {value:.4f}')
    return {'statistic': result[0], 'pvalue': result[1], 'critical_values': result[4]}

# Funcin para graficar ACF y PACF
def plot_acf_pacf(series, stock_name, lags=20):
    """
    Genera grficos ACF y PACF para sugerir rdenes de modelos y los muestra en Colab.
    """
    fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(12, 8))
    acf_vals = acf(series.dropna(), nlags=lags)
    pacf_vals = pacf(series.dropna(), nlags=lags)

    ax1.stem(range(len(acf_vals)), acf_vals)
    ax1.set_title(f'ACF - {stock_name}')
    ax1.set_xlabel('Lag')
    ax1.set_ylabel('Autocorrelation')

    ax2.stem(range(len(pacf_vals)), pacf_vals)
    ax2.set_title(f'PACF - {stock_name}')
    ax2.set_xlabel('Lag')
    ax2.set_ylabel('Partial Autocorrelation')

    plt.tight_layout()
    plt.savefig(f'{stock_name}_acf_pacf.png')
    display(Image(filename=f'{stock_name}_acf_pacf.png'))
    plt.close()

# Funcin para ajustar modelos AR, ARMA y ARIMA
def fit_models(series, stock_name):
    """
    Ajusta modelos AR, ARMA y ARIMA y selecciona el mejor basado en AIC/BIC.
    Returns:
        dict: Mejor modelo y mtricas
    """
    models = {}

    # AR(1)
    ar_model = AutoReg(series.dropna(), lags=1).fit()
    models['AR(1)'] = {'model': ar_model, 'aic': ar_model.aic, 'bic': ar_model.bic}

    # ARMA(1,1)
    arma_model = ARIMA(series.dropna(), order=(1,0,1)).fit()
    models['ARMA(1,1)'] = {'model': arma_model, 'aic': arma_model.aic, 'bic': arma_model.bic}

    # ARIMA(1,1,1)
    arima_model = ARIMA(series.dropna(), order=(1,1,1)).fit()
    models['ARIMA(1,1,1)'] = {'model': arima_model, 'aic': arima_model.aic, 'bic': arima_model.bic}

    # Seleccionar el mejor modelo basado en AIC
    best_model = min(models.items(), key=lambda x: x[1]['aic'])

    print(f'\nModel Comparison for {stock_name}:')
    for name, info in models.items():
        print(f'{name}: AIC={info["aic"]:.2f}, BIC={info["bic"]:.2f}')
    print(f'Best Model: {best_model[0]} (AIC={best_model[1]["aic"]:.2f})')

    return best_model

# Funcin para graficar ajuste y pronstico
def plot_forecast(series, model_info, stock_name, forecast_steps=10):
    """
    Grafica el ajuste del modelo y el pronstico, y lo muestra en Colab.
    """
    model = model_info[1]['model']
    model_name = model_info[0]

    # Generar pronstico
    if model_name.startswith('AR'):
        forecast = model.predict(start=len(series), end=len(series)+forecast_steps-1)
        fitted = model.fittedvalues
        # Alinear series con fitted values (excluir primeros lags)
        fitted_index = fitted.index
        series_aligned = series.loc[fitted_index]
    else:
        forecast = model.forecast(steps=forecast_steps)
        fitted = model.fittedvalues()
        # Para ARMA/ARIMA, usar el ndice de fitted values
        fitted_index = fitted.index
        series_aligned = series.loc[fitted_index]

    # Crear ndice para pronstico
    last_date = series.index[-1]
    forecast_index = pd.date_range(start=last_date, periods=forecast_steps+1, freq='5min')[1:]

    plt.figure(figsize=(12, 6))
    plt.plot(series.index, series, label='Observed', color='blue')
    plt.plot(fitted_index, fitted, label='Fitted', color='green', linestyle='--')
    plt.plot(forecast_index, forecast, label='Forecast', color='red', linestyle='--')
    plt.title(f'{stock_name} - {model_name} Fit and Forecast')
    plt.xlabel('Time')
    plt.ylabel('Price')
    plt.legend()
    plt.grid(True)
    plt.savefig(f'{stock_name}_forecast.png')
    display(Image(filename=f'{stock_name}_forecast.png'))
    plt.close()
